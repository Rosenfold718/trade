    89→    89→export function SMA(data: number[], period: number): number[] {
    90→    90→  const result: number[] = [];
    91→    91→  for (let i = 0; i < data.length; i++) {
    92→    92→    if (i < period - 1) { result.push(NaN); continue; }
    93→    93→    let sum = 0;
    94→    94→    for (let j = i - period + 1; j <= i; j++) { sum += data[j]; }
    95→    95→    result.push(sum / period);
    96→    96→  }
    97→    97→  return result;
    98→    98→}

   100→   100→export function EMA(data: number[], period: number): number[] {
   101→   101→  const result: number[] = [];
   102→   102→  const multiplier = 2 / (period + 1);
   103→   103→  let sum = 0;
   104→   104→  for (let i = 0; i < period; i++) {
   105→   105→    sum += data[i];
   106→   106→    result.push(NaN);
   107→   107→  }
   108→   108→  result[period - 1] = sum / period;
   109→   109→  for (let i = period; i < data.length; i++) {
   110→   110→    result.push((data[i] - result[i - 1]) * multiplier + result[i - 1]);
   111→   111→  }
   112→   112→  return result;
   113→   113→}

   115→   115→export function RSI(closes: number[], period: number = 14): number[] {
   116→   116→  const result: number[] = [];
   117→   117→  const gains: number[] = [];
   118→   118→  const losses: number[] = [];
   119→   119→  for (let i = 0; i < closes.length; i++) {
   120→   120→    if (i === 0) { gains.push(0); losses.push(0); result.push(NaN); continue; }
   121→   121→    const change = closes[i] - closes[i - 1];
   122→   122→    gains.push(change > 0 ? change : 0);
   123→   123→    losses.push(change < 0 ? Math.abs(change) : 0);
   124→   124→    if (i < period) { result.push(NaN); continue; }
   125→   125→    let avgGain = gains.slice(i - period + 1, i + 1).reduce((a, b) => a + b, 0) / period;
   126→   126→    let avgLoss = losses.slice(i - period + 1, i + 1).reduce((a, b) => a + b, 0) / period;
   127→   127→    if (avgLoss === 0) { result.push(100); } else { result.push(100 - 100 / (1 + avgGain / avgLoss)); }
   128→   128→  }
   129→   129→  return result;
   130→   130→}

   132→   132→export function MACD(closes: number[], fastPeriod: number = 12, slowPeriod: number = 26, signalPeriod: number = 9): { macd: number[]; signal: number[]; histogram: number[] } {
   133→   133→  const fastEMA = EMA(closes, fastPeriod);
   134→   134→  const slowEMA = EMA(closes, slowPeriod);
   135→   135→  const macdLine: number[] = [];
   136→   136→  for (let i = 0; i < closes.length; i++) {
   137→   137→    macdLine.push(isNaN(fastEMA[i]) || isNaN(slowEMA[i]) ? NaN : fastEMA[i] - slowEMA[i]);
   138→   138→  }
   139→   139→  const validMacd = macdLine.filter(v => !isNaN(v));
   140→   140→  const signalLine = EMA(validMacd, signalPeriod);
   141→   141→  const fullSignal: number[] = [];
   142→   142→  let vIdx = 0;
   143→   143→  for (let i = 0; i < macdLine.length; i++) {
   144→   144→    if (isNaN(macdLine[i])) { fullSignal.push(NaN); } else { fullSignal.push(signalLine[vIdx] || NaN); vIdx++; }
   145→   145→  }
   146→   146→  const histogram: number[] = macdLine.map((m, i) => isNaN(m) || isNaN(fullSignal[i]) ? NaN : m - fullSignal[i]);
   147→   147→  return { macd: macdLine, signal: fullSignal, histogram };
   148→   148→}

   150→   150→export function BollingerBands(closes: number[], period: number = 20, stdDev: number = 2): { upper: number[]; middle: number[]; lower: number[] } {
   151→   151→  const middle = SMA(closes, period);
   152→   152→  const upper: number[] = [];
   153→   153→  const lower: number[] = [];
   154→   154→  for (let i = 0; i < closes.length; i++) {
   155→   155→    if (isNaN(middle[i])) { upper.push(NaN); lower.push(NaN); continue; }
   156→   156→    let sumSq = 0;
   157→   157→    for (let j = i - period + 1; j <= i; j++) { sumSq += Math.pow(closes[j] - middle[i], 2); }
   158→   158→    const sd = Math.sqrt(sumSq / period);
   159→   159→    upper.push(middle[i] + stdDev * sd);
   160→   160→    lower.push(middle[i] - stdDev * sd);
   161→   161→  }
   162→   162→  return { upper, middle, lower };
   163→   163→}

   165→   165→export function Stochastic(highs: number[], lows: number[], closes: number[], kPeriod: number = 14, dPeriod: number = 3): { k: number[]; d: number[] } {
   166→   166→  const k: number[] = [];
   167→   167→  for (let i = 0; i < closes.length; i++) {
   168→   168→    if (i < kPeriod - 1) { k.push(NaN); continue; }
   169→   169→    let hh = -Infinity, ll = Infinity;
   170→   170→    for (let j = i - kPeriod + 1; j <= i; j++) { if (highs[j] > hh) hh = highs[j]; if (lows[j] < ll) ll = lows[j]; }
   171→   171→    k.push(hh === ll ? 50 : ((closes[i] - ll) / (hh - ll)) * 100);
   172→   172→  }
   173→   173→  const validK = k.filter(v => !isNaN(v));
   174→   174→  const dRaw = SMA(validK, dPeriod);
   175→   175→  const d: number[] = [];
   176→   176→  let vi = 0;
   177→   177→  for (let i = 0; i < k.length; i++) { if (isNaN(k[i])) { d.push(NaN); } else { d.push(dRaw[vi] || NaN); vi++; } }
   178→   178→  return { k, d };
   179→   179→}

   181→   181→export function ATR(highs: number[], lows: number[], closes: number[], period: number = 14): number[] {
   182→   182→  const tr: number[] = [];
   183→   183→  const result: number[] = [];
   184→   184→  for (let i = 0; i < closes.length; i++) {
   185→   185→    if (i === 0) tr.push(highs[i] - lows[i]);
   186→   186→    else tr.push(Math.max(highs[i] - lows[i], Math.abs(highs[i] - closes[i - 1]), Math.abs(lows[i] - closes[i - 1])));
   187→   187→  }
   188→   188→  for (let i = 0; i < tr.length; i++) {
   189→   189→    if (i < period - 1) result.push(NaN);
   190→   190→    else if (i === period - 1) result.push(tr.slice(0, period).reduce((a, b) => a + b, 0) / period);
   191→   191→    else result.push((result[i - 1] * (period - 1) + tr[i]) / period);
   192→   192→  }
   193→   193→  return result;
   194→   194→}

   196→   196→export function VWAP(data: OHLCV[]): number[] {
   197→   197→  const result: number[] = [];
   198→   198→  let cumTPV = 0, cumVol = 0;
   199→   199→  for (const c of data) {
   200→   200→    const tp = (c.high + c.low + c.close) / 3;
   201→   201→    cumTPV += tp * c.volume;
   202→   202→    cumVol += c.volume;
   203→   203→    result.push(cumVol === 0 ? c.close : cumTPV / cumVol);
   204→   204→  }
   205→   205→  return result;
   206→   206→}

   208→   208→export function ADX(highs: number[], lows: number[], closes: number[], period: number = 14): number[] {
   209→   209→  const plusDM: number[] = [], minusDM: number[] = [], tr: number[] = [];
   210→   210→  for (let i = 0; i < closes.length; i++) {
   211→   211→    if (i === 0) { plusDM.push(0); minusDM.push(0); tr.push(highs[i] - lows[i]); continue; }
   212→   212→    const up = highs[i] - highs[i - 1], down = lows[i - 1] - lows[i];
   213→   213→    plusDM.push(up > down && up > 0 ? up : 0);
   214→   214→    minusDM.push(down > up && down > 0 ? down : 0);
   215→   215→    tr.push(Math.max(highs[i] - lows[i], Math.abs(highs[i] - closes[i - 1]), Math.abs(lows[i] - closes[i - 1])));
   216→   216→  }
   217→   217→  const sTR = smoothWilder(tr, period), sPDM = smoothWilder(plusDM, period), sMDM = smoothWilder(minusDM, period);
   218→   218→  const plusDI: number[] = [], minusDI: number[] = [], dx: number[] = [];
   219→   219→  for (let i = 0; i < closes.length; i++) {
   220→   220→    plusDI.push(!isNaN(sTR[i]) && sTR[i] !== 0 ? (sPDM[i] / sTR[i]) * 100 : 0);
   221→   221→    minusDI.push(!isNaN(sTR[i]) && sTR[i] !== 0 ? (sMDM[i] / sTR[i]) * 100 : 0);
   222→   222→    const diSum = plusDI[i] + minusDI[i];
   223→   223→    dx.push(diSum === 0 ? 0 : (Math.abs(plusDI[i] - minusDI[i]) / diSum) * 100);
   224→   224→  }
   225→   225→  return smoothWilder(dx, period);
   226→   226→}

   228→   228→function smoothWilder(data: number[], period: number): number[] {
   229→   229→  const result: number[] = [];
   230→   230→  for (let i = 0; i < data.length; i++) {
   231→   231→    if (i < period - 1) result.push(NaN);
   232→   232→    else if (i === period - 1) result.push(data.slice(0, period).reduce((a, b) => a + b, 0) / period);
   233→   233→    else result.push(!isNaN(result[i - 1]) ? (result[i - 1] * (period - 1) + data[i]) / period : data[i]);
   234→   234→  }
   235→   235→  return result;
   236→   236→}

   238→   238→export function OBV(closes: number[], volumes: number[]): number[] {
   239→   239→  const result: number[] = [0];
   240→   240→  for (let i = 1; i < closes.length; i++) {
   241→   241→    if (closes[i] > closes[i - 1]) result.push(result[i - 1] + volumes[i]);
   242→   242→    else if (closes[i] < closes[i - 1]) result.push(result[i - 1] - volumes[i]);
   243→   243→    else result.push(result[i - 1]);
   244→   244→  }
   245→   245→  return result;
   246→   246→}

   459→   459→/** Find the most recent swing low — a candle whose low is lower than N candles on each side */
   460→   460→export function findLastSwingLow(data: OHLCV[], lookback: number = 5): number {
   461→   461→  if (data.length < lookback * 2 + 1) {
   462→   462→    return data.length > 0 ? Math.min(...data.slice(-20).map(d => d.low)) : 0;
   463→   463→  }
   464→   464→  const recent = data.slice(-(lookback * 4)); // Look in the last ~20 candles
   465→   465→  for (let i = recent.length - 1; i >= lookback; i--) {
   466→   466→    let isSwing = true;
   467→   467→    for (let j = 1; j <= lookback; j++) {
   468→   468→      if (recent[i].low >= recent[i - j].low || recent[i].low >= recent[Math.min(i + j, recent.length - 1)].low) {
   469→   469→        isSwing = false;
   470→   470→        break;
   471→   471→      }
   472→   472→    }
   473→   473→    if (isSwing) return recent[i].low;
   474→   474→  }
   475→   475→  // Fallback: lowest low of recent data
   476→   476→  return Math.min(...data.slice(-20).map(d => d.low));
   477→   477→}

   502→   502→export function findKeyLevels(data: OHLCV[]): {
   503→   503→  supports: number[]; resistances: number[];
   504→   504→  pivot: number; nearestSupport: number; nearestResistance: number;
   505→   505→} {
   506→   506→  if (data.length < 5) return { supports: [], resistances: [], pivot: 0, nearestSupport: 0, nearestResistance: 0 };
   507→   507→
   508→   508→  const currentPrice = data[data.length - 1].close;
   509→   509→  const recent = data.slice(-50);
   510→   510→
   511→   511→  // Find swing highs and swing lows
   512→   512→  const swingHighs: number[] = [];
   513→   513→  const swingLows: number[] = [];
   514→   514→
   515→   515→  for (let i = 2; i < recent.length - 2; i++) {
   516→   516→    if (recent[i].high > recent[i - 1].high && recent[i].high > recent[i - 2].high &&
   517→   517→        recent[i].high > recent[i + 1].high && recent[i].high > recent[i + 2].high) {
   518→   518→      swingHighs.push(recent[i].high);
   519→   519→    }
   520→   520→    if (recent[i].low < recent[i - 1].low && recent[i].low < recent[i - 2].low &&
   521→   521→        recent[i].low < recent[i + 1].low && recent[i].low < recent[i + 2].low) {
   522→   522→      swingLows.push(recent[i].low);
   523→   523→    }
   524→   524→  }
   525→   525→
   526→   526→  // Cluster nearby levels (within 0.3% of each other)
   527→   527→  const clusterLevels = (levels: number[]): number[] => {
   528→   528→    if (levels.length === 0) return [];
   529→   529→    const sorted = [...levels].sort((a, b) => a - b);
   530→   530→    const clusters: number[][] = [[sorted[0]]];
   531→   531→    for (let i = 1; i < sorted.length; i++) {
   532→   532→      const lastCluster = clusters[clusters.length - 1];
   533→   533→      const avg = lastCluster.reduce((a, b) => a + b, 0) / lastCluster.length;
   534→   534→      if (Math.abs(sorted[i] - avg) / avg < 0.003) {
   535→   535→        lastCluster.push(sorted[i]);
   536→   536→      } else {
   537→   537→        clusters.push([sorted[i]]);
   538→   538→      }
   539→   539→    }
   540→   540→    return clusters.map(c => c.reduce((a, b) => a + b, 0) / c.length).sort((a, b) => a - b);
   541→   541→  };
   542→   542→
   543→   543→  const supports = clusterLevels(swingLows.filter(l => l < currentPrice)).reverse();
   544→   544→  const resistances = clusterLevels(swingHighs.filter(h => h > currentPrice));
   545→   545→
   546→   546→  const lastCandle = recent[recent.length - 1];
   547→   547→  const pivot = (lastCandle.high + lastCandle.low + lastCandle.close) / 3;
   548→   548→
   549→   549→  const nearestSupport = supports.length > 0 ? supports[0] : pivot * 0.97;
   550→   550→  const nearestResistance = resistances.length > 0 ? resistances[0] : pivot * 1.03;
   551→   551→
   552→   552→  return { supports, resistances, pivot, nearestSupport, nearestResistance };
   553→   553→}

   747→   747→    score > 3 ? 'BULLISH' : score < -3 ? 'BEARISH' : 'NEUTRAL';
   748→   748→

   750→   750→                score > 15 ? 'Восходящий' :
   751→   751→                score < -40 ? 'Сильный нисходящий' :
   752→   752→                score < -15 ? 'Нисходящий' : 'Боковик';
   753→   753→
   754→   754→  const levels = findKeyLevels(data);
   755→   755→  const keyLevel = direction === 'BULLISH'
   756→   756→    ? `Сопр: ${formatTP(levels.nearestResistance)}`

   759→   759→    : `Пивот: ${formatTP(levels.pivot)}`;
   760→   760→
   761→   761→  return { direction, score, trend, keyLevel };

   924→   924→    flatSignal.momentum = momentum;

   940→   940→  // Philosophy: A good trader NEVER chases the market. They wait for price to come to them.
   941→   941→  // KEY PRINCIPLE: Enter at a pullback level, place stop BEYOND the structure that invalidates the trade.
   942→   942→  // For LONG: enter on a pullback to EMA/Fib/VWAP, stop below the recent swing low
   943→   943→  // For SHORT: enter on a bounce to EMA/Fib/VWAP, stop above the recent swing high

   960→   960→  // Calculate pullback targets — these are the "ideal" entry zones
   961→   961→  if (direction === 'LONG') {

   973→   973→    const candidates = [

   987→   987→      entryType = 'LIMIT';

   993→   993→      entryType = 'LIMIT';
   994→   994→      entryReason = 'Лимитный ордер на откате (0.5 ATR от текущей цены)';

   998→   998→    // DO NOT use resistance as entry — resistance is where the STOP goes!
   999→   999→    const bounceToEMA = lastEma21;                         // Entry at EMA21 (dynamic resistance)
  1000→  1000→    const bounceToVWAP = volumeAnalysis.vwap;              // Entry at VWAP (fair value)
  1001→  1001→    const fib382 = currentPrice + (recentHigh - recentLow) * 0.382;  // 38.2% retracement up
  1002→  1002→    const fib50 = currentPrice + (recentHigh - recentLow) * 0.5;     // 50% retracement up

  1004→  1004→

  1006→  1006→      { price: bounceToEMA, reason: 'Откат к EMA21 — динамическое сопротивление', minDist: 0.001 },

  1028→  1028→  const minSlPct = 0.01; // 1.0% minimum SL distance for crypto — 0.5% gets hit by normal volatility
  1029→  1029→  let stopLoss: number;
  1030→  1030→  if (direction === 'LONG') {

  1038→  1038→      Math.min(structuralStop, supportStop),
  1039→  1039→      atrStop,
  1040→  1040→      minPctStop                                      // Enforce minimum distance
  1041→  1041→    );

  1043→  1043→    // Final safety: ensure at least 1.0% distance

  1045→  1045→  } else {
  1046→  1046→    // For SHORT: stop above the last swing high + buffer
  1047→  1047→    const structuralStop = swingHigh + atr * 0.3;
  1048→  1048→    const atrStop = entry + atr * 2.0;

  1053→  1053→      Math.max(structuralStop, resistanceStop),
  1054→  1054→      atrStop,
  1055→  1055→      minPctStop
  1056→  1056→    );
  1057→  1057→    if (stopLoss - entry < atr * 1.0) stopLoss = entry + atr * 1.5;
  1058→  1058→    if ((stopLoss - entry) / entry < minSlPct) stopLoss = entry * (1 + minSlPct);
  1059→  1059→  }

  1062→  1062→  const risk = Math.abs(entry - stopLoss);
  1063→  1063→  const takeProfit1 = direction === 'LONG' ? entry + risk * 1.5 : entry - risk * 1.5;
  1064→  1064→  const takeProfit2 = direction === 'LONG' ? entry + risk * 2.5 : entry - risk * 2.5;
  1065→  1065→  const takeProfit3 = direction === 'LONG' ? entry + risk * 4 : entry - risk * 4;
  1066→  1066→
  1067→  1067→  const reward = Math.abs(takeProfit2 - entry);
  1068→  1068→  const riskReward = risk > 0 ? reward / risk : 0;
  1069→  1069→
  1070→  1070→  // Hold duration based on timeframe
  1071→  1071→  const holdDurationHours: number =
  1072→  1072→    timeframe === '1m' ? 0.5 :
  1073→  1073→    timeframe === '5m' ? 1 :
  1074→  1074→    timeframe === '15m' ? 3 :
  1075→  1075→    timeframe === '1h' ? 8 :
  1076→  1076→    timeframe === '4h' ? 24 :
  1077→  1077→    12;
  1078→  1078→
  1079→  1079→  const holdDuration = holdDurationHours <= 0.5 ? `${holdDurationHours * 60}-${holdDurationHours * 60 + 15} мин` :
  1080→  1080→                       holdDurationHours <= 2 ? `${holdDurationHours * 60 - 15}-${holdDurationHours * 60 + 30} мин` :
  1081→  1081→                       holdDurationHours <= 6 ? `${holdDurationHours - 1}-${holdDurationHours + 2} ч` :
  1082→  1082→                       holdDurationHours <= 12 ? `${holdDurationHours - 2}-${holdDurationHours + 4} ч` :
  1083→  1083→                       `${Math.round(holdDurationHours / 24 * 10) / 10}-${Math.round((holdDurationHours / 24 + 1) * 10) / 10} дня`;
  1084→  1084→
  1085→  1085→  // Confidence calculation — IMPROVED
  1086→  1086→  let confidence = 0;
  1087→  1087→  confidence += Math.min(35, Math.abs(adjustedScore) * 0.45); // TF score
  1088→  1088→  confidence += trend === (direction === 'LONG' ? 'BULLISH' : 'BEARISH') ? 15 : trend === 'SIDEWAYS' ? 5 : 0;
  1089→  1089→  confidence += momentum === 'STRONG' ? 15 : momentum === 'MODERATE' ? 8 : 0;
  1090→  1090→  confidence += higherTFVerdict ? (higherTFVerdict.direction === (direction === 'LONG' ? 'BULLISH' : 'BEARISH') ? 15 : 0) : 5;
  1091→  1091→  confidence += riskReward >= 3 ? 10 : riskReward >= 2 ? 5 : 0;
  1092→  1092→  // Bonus for candle pattern
  1093→  1093→  if (candlePattern && candlePattern.signal === (direction === 'LONG' ? 'BULLISH' : 'BEARISH')) {
  1094→  1094→    confidence += candlePattern.strength * 1.5;
  1095→  1095→  }
  1096→  1096→  // Bonus for volume confirmation
  1097→  1097→  if (volumeAnalysis.isVolumeSpike && volumeAnalysis.signal?.includes(direction === 'LONG' ? 'BULLISH' : 'BEARISH')) {
  1098→  1098→    confidence += 8;
  1099→  1099→  }
  1100→  1100→  // Bonus for VWAP confirmation
  1101→  1101→  if (direction === 'LONG' && volumeAnalysis.vwapPosition === 'ABOVE') confidence += 5;
  1102→  1102→  if (direction === 'SHORT' && volumeAnalysis.vwapPosition === 'BELOW') confidence += 5;
  1103→  1103→
  1104→  1104→  confidence = Math.min(95, Math.max(10, Math.round(confidence)));
  1105→  1105→
  1106→  1106→  // Build reasons and warnings
