   212тЖТ   212тЖТfunction formatPrice(price: number): string {
   213тЖТ   213тЖТ  if (!price || !isFinite(price)) return '0.00';
   214тЖТ   214тЖТ  if (price >= 1000) return price.toLocaleString('en-US', { minimumFractionDigits: 2, maximumFractionDigits: 2 });
   215тЖТ   215тЖТ  if (price >= 1) return price.toLocaleString('en-US', { minimumFractionDigits: 2, maximumFractionDigits: 4 });
   216тЖТ   216тЖТ  if (price >= 0.01) return price.toFixed(4);
   217тЖТ   217тЖТ  if (price >= 0.0001) return price.toFixed(6);
   218тЖТ   218тЖТ  return price.toFixed(8);
   219тЖТ   219тЖТ}

   221тЖТ   221тЖТfunction formatNumber(num: number): string {

   224тЖТ   224тЖТ  if (num >= 1e6) return `$${(num / 1e6).toFixed(2)}M`;
   225тЖТ   225тЖТ  return `$${num.toLocaleString()}`;

   242тЖТ   242тЖТ// Candlestick Chart
   243тЖТ   243тЖТfunction CandlestickChartComponent({ data, overlayIndicators, positionTool, tradeSignal }: {
   244тЖТ   244тЖТ  data: ChartDataPoint[]; overlayIndicators: boolean; positionTool: PositionTool | null; tradeSignal: TradeSignal | null;
   245тЖТ   245тЖТ}) {
   246тЖТ   246тЖТ  if (data.length === 0) return null;
   247тЖТ   247тЖТ  const width = 960, height = 480;
   248тЖТ   248тЖТ  const margin = { top: 20, right: 70, bottom: 30, left: 10 };
   249тЖТ   249тЖТ  const chartW = width - margin.left - margin.right;
   250тЖТ   250тЖТ  const chartH = height - margin.top - margin.bottom;
   251тЖТ   251тЖТ  const prices = data.flatMap(d => [d.high, d.low]).filter(v => v > 0);
   252тЖТ   252тЖТ  if (positionTool?.enabled) { prices.push(positionTool.entryPrice, positionTool.targetPrice, positionTool.stopLoss); }
   253тЖТ   253тЖТ  if (tradeSignal && tradeSignal.direction !== 'FLAT') { prices.push(tradeSignal.entry, tradeSignal.stopLoss, tradeSignal.takeProfit2, tradeSignal.takeProfit1); }
   254тЖТ   254тЖТ  const minP = Math.min(...prices), maxP = Math.max(...prices), range = maxP - minP || 1, padding = range * 0.08;
   255тЖТ   255тЖТ  const yMin = minP - padding, yMax = maxP + padding, yRange = yMax - yMin;
   256тЖТ   256тЖТ  const yScale = (v: number) => margin.top + chartH - ((v - yMin) / yRange) * chartH;
   257тЖТ   257тЖТ  const xScale = (i: number) => margin.left + (i / (data.length - 1 || 1)) * chartW;
   258тЖТ   258тЖТ  const candleW = Math.max(2, Math.min(20, (chartW / data.length) * 0.6));
   259тЖТ   259тЖТ  const formatY = (v: number) => {
   260тЖТ   260тЖТ    if (v >= 10000) return `$${(v/1000).toFixed(1)}k`;
   261тЖТ   261тЖТ    if (v >= 1000) return `$${v.toLocaleString('en-US', {maximumFractionDigits: 0})}`;
   262тЖТ   262тЖТ    if (v >= 1) return `$${v.toFixed(2)}`;
   263тЖТ   263тЖТ    if (v >= 0.01) return `$${v.toFixed(4)}`;
   264тЖТ   264тЖТ    return `$${v.toFixed(6)}`;
   265тЖТ   265тЖТ  };
   266тЖТ   266тЖТ  const ema9Data = data.map((d, i) => d.ema9 ? { x: xScale(i), y: yScale(d.ema9) } : null).filter(Boolean);
   267тЖТ   267тЖТ  const ema21Data = data.map((d, i) => d.ema21 ? { x: xScale(i), y: yScale(d.ema21) } : null).filter(Boolean);
   268тЖТ   268тЖТ  const linePath = (points: { x: number; y: number }[]) => points.map((p, i) => `${i === 0 ? 'M' : 'L'}${p.x},${p.y}`).join(' ');
   269тЖТ   269тЖТ  const ticks = Array.from({ length: 6 }, (_, i) => yMin + (yRange * i) / 5);
   270тЖТ   270тЖТ  const ts = tradeSignal, pt = positionTool, hasPosition = pt?.enabled && pt.entryPrice > 0;
   271тЖТ   271тЖТ  const targetPnl = hasPosition ? (pt!.direction === 'LONG' ? ((pt!.targetPrice - pt!.entryPrice) / pt!.entryPrice) : ((pt!.entryPrice - pt!.targetPrice) / pt!.entryPrice)) * pt!.amount * pt!.leverage : 0;
   272тЖТ   272тЖТ  const stopPnl = hasPosition ? (pt!.direction === 'LONG' ? ((pt!.stopLoss - pt!.entryPrice) / pt!.entryPrice) : ((pt!.entryPrice - pt!.stopLoss) / pt!.entryPrice)) * pt!.amount * pt!.leverage : 0;
   273тЖТ   273тЖТ  const rr = hasPosition && stopPnl !== 0 ? Math.abs(targetPnl / stopPnl) : 0;
   274тЖТ   274тЖТ
   275тЖТ   275тЖТ  const LevelLine = ({ price, color, label, opacity = 1 }: { price: number; color: string; label: string; opacity?: number }) => {
   276тЖТ   276тЖТ    if (price <= 0) return null; const y = yScale(price); if (y < margin.top || y > margin.top + chartH) return null;
   277тЖТ   277тЖТ    const priceLabel = formatPrice(price);
   278тЖТ   278тЖТ    const labelW = Math.max(62, (label.length + priceLabel.length + 3) * 5.5);
   279тЖТ   279тЖТ    return <><line x1={margin.left} y1={y} x2={width - margin.right} y2={y} stroke={color} strokeWidth={1.2} strokeDasharray="6 3" opacity={opacity} /><rect x={width - margin.right + 2} y={y - 8} width={labelW} height={16} rx={3} fill={color} opacity={opacity} /><text x={width - margin.right + 2 + labelW/2} y={y + 3} fontSize={7} fill="white" fontWeight="bold" textAnchor="middle" opacity={opacity}>{label} ${priceLabel}</text></>;
   280тЖТ   280тЖТ  };
   281тЖТ   281тЖТ
   282тЖТ   282тЖТ  return (
   283тЖТ   283тЖТ    <svg width="100%" viewBox={`0 0 ${width} ${height}`} className="w-full" preserveAspectRatio="xMidYMid meet">
   284тЖТ   284тЖТ      {ticks.map((t, i) => (<g key={i}><line x1={margin.left} y1={yScale(t)} x2={width - margin.right} y2={yScale(t)} stroke="hsl(var(--border))" strokeWidth={0.5} opacity={0.3} /><text x={width - margin.right + 5} y={yScale(t) + 4} fontSize={9} fill="hsl(var(--muted-foreground))" textAnchor="start">{formatY(t)}</text></g>))}
   285тЖТ   285тЖТ      {data.filter((_, i) => i % Math.max(1, Math.floor(data.length / 8)) === 0).map((d, i) => { const idx = data.indexOf(d); return <text key={i} x={xScale(idx)} y={height - 5} fontSize={8} fill="hsl(var(--muted-foreground))" textAnchor="middle">{d.time || d.date.slice(0, 5)}</text>; })}
   286тЖТ   286тЖТ      {/* Signal zones */}
   287тЖТ   287тЖТ      {ts && ts.direction !== 'FLAT' && <><rect x={margin.left} y={yScale(Math.max(ts.entry, ts.takeProfit2))} width={chartW} height={Math.abs(yScale(ts.entry) - yScale(ts.takeProfit2))} fill={ts.direction === 'LONG' ? '#10b981' : '#ef4444'} opacity={0.04} /><rect x={margin.left} y={yScale(Math.max(ts.entry, ts.stopLoss))} width={chartW} height={Math.abs(yScale(ts.entry) - yScale(ts.stopLoss))} fill="#ef4444" opacity={0.06} /></>}
   288тЖТ   288тЖТ      {/* Candles */}
   289тЖТ   289тЖТ      {data.map((d, i) => { const isGreen = d.close >= d.open; const color = isGreen ? '#10b981' : '#ef4444'; const bodyTop = yScale(Math.max(d.open, d.close)); const bodyBottom = yScale(Math.min(d.open, d.close)); const cx = xScale(i); return <g key={i}><line x1={cx} y1={yScale(d.high)} x2={cx} y2={yScale(d.low)} stroke={color} strokeWidth={1} /><rect x={cx - candleW / 2} y={bodyTop} width={candleW} height={Math.max(1, bodyBottom - bodyTop)} fill={color} stroke={color} strokeWidth={0.5} opacity={0.9} rx={1} /></g>; })}
   290тЖТ   290тЖТ      {/* EMA overlays */}
   291тЖТ   291тЖТ      {overlayIndicators && ema9Data.length > 1 && <path d={linePath(ema9Data)} fill="none" stroke="#f97316" strokeWidth={1.5} strokeDasharray="4 2" />}

   354тЖТ   354тЖТfunction DepositChart({ data, initialDeposit, totalDebt }: { data: DepositSnapshot[]; initialDeposit: number; totalDebt?: number }) {
   355тЖТ   355тЖТ  if (data.length < 2) return null;
   356тЖТ   356тЖТ  const w = 560, h = 100, margin = { top: 10, right: 50, bottom: 10, left: 10 };
   357тЖТ   357тЖТ  const cw = w - margin.left - margin.right, ch = h - margin.top - margin.bottom;
   358тЖТ   358тЖТ  // Use equity (includes locked positions + unrealized PnL)
   359тЖТ   359тЖТ  const balances = data.map(d => d.equity);
   360тЖТ   360тЖТ  // Reference: effective capital = initialDeposit + totalDebt (what was actually available)
   361тЖТ   361тЖТ  const effectiveCapital = initialDeposit + (totalDebt || 0);
   362тЖТ   362тЖТ  const minB = Math.min(...balances, effectiveCapital) * 0.95;
   363тЖТ   363тЖТ  const maxB = Math.max(...balances, effectiveCapital) * 1.05;
   364тЖТ   364тЖТ  const range = maxB - minB || 1;
   365тЖТ   365тЖТ  const xS = (i: number) => margin.left + (i / (data.length - 1)) * cw;
   366тЖТ   366тЖТ  const yS = (v: number) => margin.top + ch - ((v - minB) / range) * ch;
   367тЖТ   367тЖТ  // Profit = equity above effective capital
   368тЖТ   368тЖТ  const lastVal = data[data.length - 1].equity;
   369тЖТ   369тЖТ  const isProfit = lastVal >= effectiveCapital;
   370тЖТ   370тЖТ  const lineColor = isProfit ? '#10b981' : '#ef4444';
   371тЖТ   371тЖТ  const areaPath = data.map((d, i) => `${i === 0 ? 'M' : 'L'}${xS(i)},${yS(d.equity)}`).join(' ') + ` L${xS(data.length - 1)},${margin.top + ch} L${margin.left},${margin.top + ch} Z`;
   372тЖТ   372тЖТ  const linePath = data.map((d, i) => `${i === 0 ? 'M' : 'L'}${xS(i)},${yS(d.equity)}`).join(' ');
   373тЖТ   373тЖТ  // PnL relative to own capital only (not debt)
   374тЖТ   374тЖТ  const ownEquity = lastVal - (totalDebt || 0);
   375тЖТ   375тЖТ  const pnlPct = ((ownEquity / initialDeposit - 1) * 100).toFixed(1);
   376тЖТ   376тЖТ  return (
   377тЖТ   377тЖТ    <svg width="100%" viewBox={`0 0 ${w} ${h}`} className="w-full" preserveAspectRatio="xMidYMid meet">
   378тЖТ   378тЖТ      {/* Effective capital reference line */}
   379тЖТ   379тЖТ      <line x1={margin.left} y1={yS(effectiveCapital)} x2={margin.left + cw} y2={yS(effectiveCapital)} stroke="hsl(var(--muted-foreground))" strokeWidth={0.5} strokeDasharray="4 2" opacity={0.5} />
   380тЖТ   380тЖТ      <text x={margin.left + cw + 3} y={yS(effectiveCapital) + 3} fontSize={7} fill="hsl(var(--muted-foreground))">${effectiveCapital.toFixed(0)}</text>
   381тЖТ   381тЖТ      {/* Area */}
