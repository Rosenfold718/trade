   423‚Üí   423‚Üí
   424‚Üí   424‚Üíexport async function getTradingStats(): Promise<TradingStats> {
   425‚Üí   425‚Üí  const state = await getTraderState();
   426‚Üí   426‚Üí  const resolvedTrades = await db.trade.findMany({
   427‚Üí   427‚Üí    where: { resolved: true, pnlUSDT: { not: null } },
   428‚Üí   428‚Üí    orderBy: { closedAt: 'asc' },
   429‚Üí   429‚Üí  });
   430‚Üí   430‚Üí
   431‚Üí   431‚Üí  const wins = resolvedTrades.filter(t => t.result === 'WIN' && t.pnlUSDT !== null);
   432‚Üí   432‚Üí  const losses = resolvedTrades.filter(t => t.result === 'LOSS' && t.pnlUSDT !== null);
   433‚Üí   433‚Üí
   434‚Üí   434‚Üí  const totalWins = wins.length;
   435‚Üí   435‚Üí  const totalLosses = losses.length;
   436‚Üí   436‚Üí  const totalPnl = state.totalPnl;
   437‚Üí   437‚Üí  const totalTrades = resolvedTrades.length;
   438‚Üí   438‚Üí
   439‚Üí   439‚Üí  const winRate = totalTrades > 0 ? (totalWins / totalTrades) * 100 : 0;
   440‚Üí   440‚Üí
   441‚Üí   441‚Üí  const winPnls = wins.map(t => t.pnlUSDT!);
   442‚Üí   442‚Üí  const lossPnls = losses.map(t => t.pnlUSDT!);
   443‚Üí   443‚Üí
   444‚Üí   444‚Üí  const avgWin = winPnls.length > 0 ? winPnls.reduce((a, b) => a + b, 0) / winPnls.length : 0;
   445‚Üí   445‚Üí  const avgLoss = lossPnls.length > 0 ? lossPnls.reduce((a, b) => a + b, 0) / lossPnls.length : 0;
   446‚Üí   446‚Üí
   447‚Üí   447‚Üí  const avgPnl = totalTrades > 0 ? totalPnl / totalTrades : 0;
   448‚Üí   448‚Üí
   449‚Üí   449‚Üí  // Profit factor: gross profits / gross losses (absolute)
   450‚Üí   450‚Üí  const grossProfit = winPnls.reduce((a, b) => a + b, 0);
   451‚Üí   451‚Üí  const grossLoss = Math.abs(lossPnls.reduce((a, b) => a + b, 0));
   452‚Üí   452‚Üí  const profitFactor = grossLoss > 0 ? grossProfit / grossLoss : grossProfit > 0 ? Infinity : 0;
   453‚Üí   453‚Üí
   454‚Üí   454‚Üí  // Expectancy: (winRate% * avgWin) - (lossRate% * |avgLoss|)
   455‚Üí   455‚Üí  const lossRate = totalTrades > 0 ? (totalLosses / totalTrades) : 0;
   456‚Üí   456‚Üí  const expectancy = (winRate / 100 * avgWin) - (lossRate * Math.abs(avgLoss));
   457‚Üí   457‚Üí
   458‚Üí   458‚Üí  // Max drawdown from deposit snapshots
   459‚Üí   459‚Üí  const snapshots = await db.depositSnapshot.findMany({ orderBy: { timestamp: 'asc' } });
   460‚Üí   460‚Üí  let maxDrawdown = 0;
   461‚Üí   461‚Üí  let peak = 0;
   462‚Üí   462‚Üí  for (const snap of snapshots) {
   463‚Üí   463‚Üí    if (snap.equity > peak) peak = snap.equity;
   464‚Üí   464‚Üí    const dd = peak > 0 ? ((peak - snap.equity) / peak) * 100 : 0;
   465‚Üí   465‚Üí    if (dd > maxDrawdown) maxDrawdown = dd;
   466‚Üí   466‚Üí  }
   467‚Üí   467‚Üí
   468‚Üí   468‚Üí  // Sharpe ratio (annualized, assuming ~6 trades/day, 365 days)
   469‚Üí   469‚Üí  const allPnls = resolvedTrades.map(t => t.pnlUSDT ?? 0);
   470‚Üí   470‚Üí  const sharpeRatio = computeSharpe(allPnls, 2190); // ~6 * 365
   471‚Üí   471‚Üí
   472‚Üí   472‚Üí  // Sortino ratio
   473‚Üí   473‚Üí  const sortinoRatio = computeSortino(allPnls, 2190);
   474‚Üí   474‚Üí
   475‚Üí   475‚Üí  // Calmar ratio: annualized return / max drawdown
   476‚Üí   476‚Üí  const annualizedReturn = totalTrades > 0 ? (totalPnl / state.initialDeposit) * (2190 / Math.max(totalTrades, 1)) * 100 : 0;
   477‚Üí   477‚Üí  const calmarRatio = maxDrawdown > 0 ? annualizedReturn / maxDrawdown : 0;
   478‚Üí   478‚Üí
   479‚Üí   479‚Üí  // Consecutive wins/losses streaks
   480‚Üí   480‚Üí  let consecutiveWins = 0;
   481‚Üí   481‚Üí  let consecutiveLosses = 0;
   482‚Üí   482‚Üí  let currentWinStreak = 0;
   483‚Üí   483‚Üí  let currentLossStreak = 0;
   484‚Üí   484‚Üí  for (const t of resolvedTrades) {
   485‚Üí   485‚Üí    if (t.result === 'WIN') {
   486‚Üí   486‚Üí      currentWinStreak++;
   487‚Üí   487‚Üí      currentLossStreak = 0;
   488‚Üí   488‚Üí      if (currentWinStreak > consecutiveWins) consecutiveWins = currentWinStreak;
   489‚Üí   489‚Üí    } else if (t.result === 'LOSS') {
   490‚Üí   490‚Üí      currentLossStreak++;
   491‚Üí   491‚Üí      currentWinStreak = 0;
   492‚Üí   492‚Üí      if (currentLossStreak > consecutiveLosses) consecutiveLosses = currentLossStreak;
   493‚Üí   493‚Üí    } else {
   494‚Üí   494‚Üí      currentWinStreak = 0;
   495‚Üí   495‚Üí      currentLossStreak = 0;
   496‚Üí   496‚Üí    }
   497‚Üí   497‚Üí  }
   498‚Üí   498‚Üí
   499‚Üí   499‚Üí  return {
   500‚Üí   500‚Üí    totalTrades,
   501‚Üí   501‚Üí    wins: totalWins,
   502‚Üí   502‚Üí    losses: totalLosses,
   503‚Üí   503‚Üí    expired: state.expired,
   504‚Üí   504‚Üí    winRate,
   505‚Üí   505‚Üí    avgPnl,
   506‚Üí   506‚Üí    totalPnl,
   507‚Üí   507‚Üí    bestTrade: state.bestTrade,
   508‚Üí   508‚Üí    worstTrade: state.worstTrade,
   509‚Üí   509‚Üí    profitFactor,
   510‚Üí   510‚Üí    sharpeRatio,
   511‚Üí   511‚Üí    sortinoRatio,
   512‚Üí   512‚Üí    maxDrawdown,
   513‚Üí   513‚Üí    calmarRatio,
   514‚Üí   514‚Üí    avgWin,
   515‚Üí   515‚Üí    avgLoss,
   516‚Üí   516‚Üí    expectancy,
   517‚Üí   517‚Üí    consecutiveWins,
   518‚Üí   518‚Üí    consecutiveLosses,
   519‚Üí   519‚Üí  };
   520‚Üí   520‚Üí}
   521‚Üí   521‚Üí
   522‚Üí   522‚Üífunction computeSharpe(returns: number[], periodsPerYear: number): number {
   523‚Üí   523‚Üí  if (returns.length < 2) return 0;
   524‚Üí   524‚Üí  const mean = returns.reduce((a, b) => a + b, 0) / returns.length;
   525‚Üí   525‚Üí  const variance = returns.reduce((acc, r) => acc + Math.pow(r - mean, 2), 0) / (returns.length - 1);
   526‚Üí   526‚Üí  const stdDev = Math.sqrt(variance);
   527‚Üí   527‚Üí  if (stdDev === 0) return 0;
   528‚Üí   528‚Üí  return (mean / stdDev) * Math.sqrt(periodsPerYear);
   529‚Üí   529‚Üí}
   530‚Üí   530‚Üí
   531‚Üí   531‚Üífunction computeSortino(returns: number[], periodsPerYear: number): number {
   532‚Üí   532‚Üí  if (returns.length < 2) return 0;
   533‚Üí   533‚Üí  const mean = returns.reduce((a, b) => a + b, 0) / returns.length;
   534‚Üí   534‚Üí  const downsideVariance = returns
   535‚Üí   535‚Üí    .filter(r => r < 0)
   536‚Üí   536‚Üí    .reduce((acc, r) => acc + Math.pow(r, 2), 0) / returns.length;
   537‚Üí   537‚Üí  const downsideDev = Math.sqrt(downsideVariance);
